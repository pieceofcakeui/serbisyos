<script>
    let currentReviewCount = 2; // Initial review count (first 2 reviews visible)
    const reviews = document.querySelectorAll('.review');
    const seeMoreLink = document.querySelector('.see-more');

    // Function to toggle between See More and See Less
    function toggleReviews() {
        const totalReviews = reviews.length;
        if (currentReviewCount === totalReviews) {
            // If showing all reviews, show only 2 reviews again
            currentReviewCount = 2;
            seeMoreLink.textContent = 'See More';
        } else {
            // Show next 5 reviews
            currentReviewCount += 5;
            if (currentReviewCount >= totalReviews) {
                currentReviewCount = totalReviews;
                seeMoreLink.textContent = 'See Less';
            } else {
                seeMoreLink.textContent = 'See Less';
            }
        }

        updateReviewVisibility();
    }

    // Function to update visibility of reviews based on currentReviewCount
    function updateReviewVisibility() {
        reviews.forEach((review, index) => {
            if (index < currentReviewCount) {
                review.style.display = 'block';
            } else {
                review.style.display = 'none';
            }
        });
    }

    // Initialize the visibility on page load
    updateReviewVisibility();
</script>
